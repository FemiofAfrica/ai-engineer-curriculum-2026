Combined API

main.py
from fastapi import FastAPI, HTTPException, Depends
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
import time, os, logging

logging.basicConfig(
  level=logging.INFO,
  format='{"time":"%(asctime)s","level":"%(levelname)s","req_id":"%(req_id)s","message":"%(message)s"}'
)

app = FastAPI(title="AI Product")
app.add_middleware(CORSMiddleware, allow_origins=["*"])

class AskRequest(BaseModel):
  question: str
  use_agent: bool = False

class AskResponse(BaseModel):
  answer: str
  sources: list[str]
  latency_ms: float
  tokens_used: int

@app.get("/health")
async def health():
  return {
    "status": "ok",
    "db": "connected",
    "api_key": os.environ.get("OPENAI_API_KEY") is not None,
    "uptime": time.time() - start_time
  }

@app.post("/ask", response_model=AskResponse)
async def ask(req: AskRequest):
  start = time.time()
  try:
    if req.use_agent:
      result = run_agent(req.question)
    else:
      result = answer_with_rag(req.question)

    return AskResponse(
      answer=result["text"],
      sources=result.get("sources", []),
      latency_ms=round((time.time() - start) * 1000, 1),
      tokens_used=result.get("tokens", 0)
    )
  except Exception as e:
    logging.error(f"Request failed: {e}")
    raise HTTPException(status_code=500, detail=str(e))

Minimal Frontend

static/index.html
<!DOCTYPE html>
<html>
<head><title>AI Q&A</title></head>
<body>
  <div id="app">
    <h1>Ask my documents</h1>
    <textarea id="q" rows="3" cols="50"></textarea><br>
    <button onclick="ask()">Ask</button>
    <div id="result"></div>
  </div>
  <script>
    async function ask() {
      const q = document.getElementById('q').value;
      const r = await fetch('/ask', {
        method: 'POST',
        headers: {'Content-Type':'application/json'},
        body: JSON.stringify({question: q})
      });
      const data = await r.json();
      document.getElementById('result').innerHTML =
        `<p><strong>Answer:</strong> ${data.answer}</p>
         <p><small>${data.latency_ms}ms | ${data.tokens_used} tokens</small></p>`;
    }
  </script>
</body></html>
