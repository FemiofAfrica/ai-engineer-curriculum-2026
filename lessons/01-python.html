# ==============================
# Functions — the AI engineer's bread and butter
# ==============================

from typing import List, Dict, Optional, Callable
import time

# Basic function with type hints
def call_llm(prompt: str, model: str = "gpt-4o-mini") -> Dict[str, str]:
    """Simulate an LLM API call. Returns a response dict."""
    return {"role": model, "content": f"Response to: {prompt[:50]}..."}

# *args and **kwargs (common in AI SDK wrappers)
def api_wrapper(endpoint: str, *args, **kwargs) -> None:
    """Flexible API wrapper pattern."""
    headers = kwargs.get("headers", {})
    timeout = kwargs.get("timeout", 30)
    print(f"Calling {endpoint} with timeout={timeout}")

# Lambda — inline functions for quick transformations
topics = ["RAG", "agents", "fine-tuning"]
capitalise = lambda x: x.upper()
print(list(map(capitalise, topics)))  # ['RAG', 'AGENTS', 'FINE-TUNING']

# Decorator — timing wrapper for profiling LLM calls
@decorator
def timer(func: Callable) -> Callable:
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"[{func.__name__}] took {elapsed:.3f}s")
        return result
    return wrapper

@timer
def fetch_data(url: str) -> str:
    return "simulated response"

fetch_data("https://api.openai.com/v1/chat/completions")
classes.py 
# ==============================
# Classes + OOP — modelling AI concepts
# ==============================

from dataclasses import dataclass, field
from typing import List, Optional
from abc import ABC, abstractmethod

# Dataclass — zero-boilerplate data container
@dataclass
class LLMResponse:
    model: str
    content: str
    tokens_used: int = 0
    metadata: Dict[str, str] = field(default_factory=dict)

    def __post_init__(self):
        if self.tokens_used < 0:
            raise ValueError("tokens_used cannot be negative")

# Property — controlled attribute access
class ModelConfig:
    def __init__(self, name: str, max_tokens: int):
        self._name = name
        self._max_tokens = max_tokens

    @property
    def max_tokens(self) -> int:
        return self._max_tokens

    @max_tokens.setter
    def max_tokens(self, value: int):
        if value < 1 or value > 128_000:
            raise ValueError("max_tokens out of range")
        self._max_tokens = value

# Abstract Base Class — contract for LLM providers
class LLMProvider(ABC):
    @abstractmethod
    def generate(self, prompt: str) -> LLMResponse:
        ...

    @abstractmethod
    def count_tokens(self, text: str) -> int:
        ...

class OpenAIProvider(LLMProvider):
    def generate(self, prompt: str) -> LLMResponse:
        return LLMResponse(model="gpt-4o", content="...", tokens_used=50)

    def count_tokens(self, text: str) -> int:
        return len(text.split()) * 2
async_demo.py 
# ==============================
# Async — the superpower every AI engineer needs
# ==============================

import asyncio
import httpx
from typing import List, Dict

# Single async function — the basic unit
async def fetch_one(client: httpx.AsyncClient, topic: str) -> Dict[str, str]:
    # In real code: call OpenAI/Anthropic here
    await asyncio.sleep(0.5)  # simulate network call
    return {"topic": topic, "summary": f"Key insights about {topic}"}

# Concurrent fan-out — the killer pattern for AI engineering
async def process_topics(topics: List[str]) -> List[Dict]:
    async with httpx.AsyncClient(timeout=30) as client:
        tasks = [fetch_one(client, topic) for topic in topics]
        results = await asyncio.gather(*tasks, return_exceptions=True)

    # Handle errors per-task (never fail silently)
    output = []
    for topic, result in zip(topics, results):
        if isinstance(result, Exception):
            output.append({"topic": topic, "error": str(result)})
        else:
            output.append(result)
    return output

# Entry point
async def main():
    topics = ["RAG pipelines", "agent orchestration",
              "embedding models", "fine-tuning strategies"]
    results = await process_topics(topics)
    for r in results:
        print(r)

if __name__ == "__main__":
    asyncio.run(main())

file_io.py 
# ==============================
# File I/O + JSON — persisting AI outputs
# ==============================

import json
from pathlib import Path
from typing import List, Dict, Any

# pathlib — the modern way (avoid os.path)
output_dir = Path("output")
output_dir.mkdir(parents=True, exist_ok=True)

# Writing JSON with proper encoding
def save_results(filepath: Path, data: List[Dict[str, Any]]) -> None:
    with open(filepath, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=2, ensure_ascii=False)

# Reading JSON with error handling
def load_results(filepath: Path) -> List[Dict[str, Any]]:
    if not filepath.exists():
        return []
    with open(filepath, "r", encoding="utf-8") as f:
        return json.load(f)

# Usage
results = [
    {"topic": "RAG", "summary": "..."},
    {"topic": "Agents", "summary": "..."},
]
save_results(output_dir / "llm_results.json", results)
loaded = load_results(output_dir / "llm_results.json")

⚠️ Common Pitfalls

These will break your production AI code. Avoid them from day one.

  • 🔥
    Blocking the event loop — Calling time.sleep() or requests.get() inside an async function blocks ALL concurrent tasks. Always use await asyncio.sleep() and httpx.AsyncClient in async code.
  • 💣
    Mutable default arguments — def fetch(topics=[]) shares the SAME list across all calls. Always use def fetch(topics=None) and default to None, then assign inside the function.
  • 📦
    Shallow venv understanding — Running pip install globally (or without an active venv) will break your system Python eventually. Always create and activate a virtual environment. Use uv for speed.
  • 👻
    Ignoring type hints — Raw dicts everywhere with no type annotations make code unreadable at scale. Use dataclasses for structured data and always annotate function signatures with proper types.
  • 🌊
    Swallowing exceptions silently — Bare except: with no logging hides bugs. Always log errors (or use return_exceptions=True in gather) and handle failures per-task, not at the global level.
